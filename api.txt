
set_physics_override(table)
Table must be like: {
	speed = 1.0, -- multiplier to default value
	jump = 1.0, -- multiplier to default value
	gravity = 1.0, -- multiplier to default value
	sneak = true, -- whether player can sneak
	sneak_glitch = true, -- whether player can use the sneak glitch
}


table:
	global playerstatus
	local player_status["players"][playerName]
		physics:
			sprint : int
			gravity : int
			speed : int
			jump : int
			sprint_jump : int
			walking : bool
			stamina : int
			stamina_max : int
			sneak : bool
			sneak_glitch : bool

		bonus: properties
			speed
				[modname] : int
			agility
				[modname] : int
			gravity
				[modname] : int
			stamina_max
				[modname] : int



functions:
	principals:
		playerstatus.update_physics(playerName) : update variables
		playerstatus.set_physics(playerName) : set player:set_physics_override()
	
	bonus:
		playerstatus.get_bonus(playerName) return table bonus
		playerstatus.set_bonus(playerName, modname, bonus) table bonus { speed, agility, gravity, stamina_max }, table can contain 1 or mores params

	sprint:
		playerstatus.set_sprinting(playerName, is_sprint) : set sprint when is_sprint is true else walk
		playerstatus.get_stamina(playerName) : get stamina
		playerstatus.set_stamina(playerName, stamina) : set stamina
		playerstatus.get_stamina_max(playerName) : get stamina maximum
		playerstatus.set_stamina_max(playerName, stamina_max) : set stamina maximum
